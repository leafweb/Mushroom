Mushroom.js چیست؟

Mushroom.js اسکریپت ساخت پالت رنگ برای برنامه‌های تحت وب است که بسیار سبک و سریع هستش.


چطور Mushroom.js را نصب کنیم؟

برای نصب و راه‌اندازی کافیه است که اسکریپت Mushroom.js را دانلود کنیم و آن را با استفاده از تگ <script> به سند خود اضافه کنیم. 

<script src="Mushroom.js"></script>

حالا کافیست تا Mushroom را فرا بخوانیم

بیشتر یاد بگیرید

در اسناد Mushroom.js با تمام ویژگی های آن آشنا شوید:

همچنین می‌توانید با ابزار زیر پالت رنگ مورد علاقه خودتان را داشته باشید:


ارائه شده توسط
Leaf Web

راهنما

نصب Mushroom.js

Mushroom.js را دانلود کنید

دانلود Mushroom.js دانلود Mushroom.min.js

Mushroom.js را به سند HTML لینک کنید

<script src="Mushroom.js"></script>

Mushroom.js را فرا بخوانید

let M = Mushroom();

یک رنگ تنظیم کنید

M.setColor("blue");
از فرمت های رنگی HSL, RGB, Hex و رنگ های استاندارد وب میتوانید استفاده کنید.

حالت تیره یا روشن ؟ 

M.setDarkmode(true); //Lightmode 
M.setDarkmode(false); // Darkmode

تغییر حالت سریع

M.toggleMode();

اضافه کردن رنگ های دلخواه

M.addColot({
   "warn": "orange",
   "info": "royalblue"
});

پاک کردن رنگ های دلخواه 

M.clearCustomColor();

پالت تصادفی

M.random();

پالت رنگ فرعی باشد یا نباشد ؟ 

M.setPalette(true);
M.setPalette(false);

پالت فرعی وابسته به حالت

M.reversePalette(true); //با تغییر حالت به تیره پالت رنگ فرعی وارون میشود 
M.reversePalette(false); // پالت رنگ فرعی ثابت می ماند

تنظیمات رنگی بیشتر 

M.setHue(34);
M.setSaturation(70);
M.setLightness(50);
این کدها همان فرمت رنگی HSL هستند.
M.setColor("hsl(34,70%,50%)");

تنظیم سطوح روشنایی پالت رنگی فرعی 

M.setParts([0,5,10,20,40,50,60,80,90,95])

تنظیم ترکیبات رنگی 

M.setColorScheme("Complementary")

تمام ترکیبات رنگی را میتوانید در صفحه Brush مشاهده کنید

یک مرحله بالاتر 

M.setLevel("pro");

در این حالت چندین رنگ دیگه به پالت رنگ اصلی افزوده میشود. مانند: error-background, secondary-serface, tertiary-outline

اضافه کردن Mushroom به بخش های خاصی 

let M1 = Mushroom("main");
let M2 = Mushroom("#Home");
let M3 = Mushroom(":root");

شما میتوانید برای هر کدام تنظیمات خاصی قرار دهید! یرای مثال: 

M1.setColor("blue");
M2.setColor("blueviolet");
M3.setColor("purple");

گرفتن مقادیر مختلف از Mushroom

M.addTo; //String
M.code; //String
M.color; //String
M.colorScheme; //String
M.customColor; //String
M.darkmode; //Boolean
M.hasPalette; //Boolean
M.hue; //Number
M.level; //"default" or "pro"
M.lightness; //Number
M.palette; //Object
M.parts; //Array
M.reversePalette; //Boolean
M.saturation; //Number
M.themeColor; //Object
M.version; //String

یک مثال جمع و جور

var mode = eval(localStorage.getItem('darkMode'));
if (mode == undefined) {
   mode = false;
   localStorage.setItem('darkMode', mode);
}
let M = Mushroom();
M.setColor('brown');
M.setDarkmode(mode);
function Darkmode() {
   M.toggleMode();
   localStorage.setItem('darkMode', M.darkmode);
}
